- **Flujo**:  
  1. El usuario escribe el nombre o escanea el código de barras.

## 2. Lógica en el Componente
```ts
busqueda = '';
productoSeleccionado: any = null;
cantidadSeleccionada = 1;
carrito: any[] = [];
total = 0;

constructor(private service: VentasHttpServices) {}

buscarProducto() {
  this.service.buscarProducto(this.busqueda).subscribe({
    next: producto => {
      if (producto) {
        this.productoSeleccionado = producto;
        this.cantidadSeleccionada = 1;
        // Mostrar modal de cantidad
        const modal = new (window as any).bootstrap.Modal(document.getElementById('cantidadModal'));
        modal.show();
      }
    },
    error: err => console.error(err)
  });
}

agregarAlCarrito() {
  const existente = this.carrito.find(p => p.id === this.productoSeleccionado.id);
  if (existente) {
    existente.cantidad += this.cantidadSeleccionada;
  } else {
    this.carrito.push({
      id: this.productoSeleccionado.id,
      nombre: this.productoSeleccionado.nombre,
      precio: this.productoSeleccionado.precio,
      cantidad: this.cantidadSeleccionada
    });
  }
  this.recalcularTotal();
}

recalcularTotal() {
  this.total = this.carrito.reduce((acc, item) => acc + item.precio * item.cantidad, 0);
}

eliminarItem(item: any) {
  this.carrito = this.carrito.filter(p => p.id !== item.id);
  this.recalcularTotal();
}
```

buscarProducto(valor: string) {
  return from(
    this.supabase.client
      .from('productos')
      .select('*')
      .or(`nombre.ilike.%${valor}%,codigo.eq.${valor}`)
      .single()
  ).pipe(map(res => res.data));
}

## 3. Lógica en el Componente
```ts
productoSeleccionado: any = null;
cantidadSeleccionada: string = '';

buscarProducto() {
  this.service.buscarProducto(this.busqueda).subscribe({
    next: producto => {
      if (producto) {
        this.productoSeleccionado = producto;
        this.cantidadSeleccionada = '';
        const modal = new (window as any).bootstrap.Modal(document.getElementById('cantidadModal'));
        modal.show();
      }
    }
  });
}

agregarNumero(num: number) {
  this.cantidadSeleccionada += num.toString();
}

borrarCantidad() {
  this.cantidadSeleccionada = this.cantidadSeleccionada.slice(0, -1);
}

confirmarCantidad() {
  const cantidad = parseInt(this.cantidadSeleccionada || '1', 10);
  const existente = this.carrito.find(p => p.id === this.productoSeleccionado.id);

  if (existente) {
    existente.cantidad += cantidad;
  } else {
    this.carrito.push({
      id: this.productoSeleccionado.id,
      nombre: this.productoSeleccionado.nombre,
      precio: this.productoSeleccionado.precio,
      cantidad
    });
  }
  this.recalcularTotal();
}

recalcularTotal() {
  this.total = this.carrito.reduce((acc, item) => acc + item.precio * item.cantidad, 0);
}
```

## 2. Componente Angular
```ts
productos: Producto[] = [];
categorias: string[] = [];
categoriaSeleccionada = 'Todas';
buscarpro = '';

ngOnInit() {
  this.service.getProductos().subscribe(res => {
    this.productos = res;
    const cats = [...new Set(res.map(p => p.categoria))];
    this.categorias = ['Todas', ...cats];
  });
}

get productosFiltrados() {
  return this.productos.filter(p => {
    const termino = this.buscarpro.toLowerCase();
    const coincideNombre = p.nombre.toLowerCase().includes(termino);
    const coincideCodigo = p.codigo?.toString().includes(termino);
    const coincideCat = this.categoriaSeleccionada === 'Todas' || p.categoria === this.categoriaSeleccionada;
    return (coincideNombre || coincideCodigo) && coincideCat;
  });
}
```
